for (int i = 0; i < playerADiceResult.length; i++) {
	    while (playerBIndex < playerBDiceResult.length && 
	           playerBDiceResult[playerBIndex] < playerADiceResult[i]) {
	        playerBIndex++;
	    }
	    win += playerBIndex;
	}

## 개선 2

```java
for (int i = 0; i < n; i++) {
	if (!used[i] && (turn == 0 || playerA[turn - 1] < i)) {
		used[i] = true;
		playerA[turn] = i;
		roop(playerA, dice, turn + 1, used);
		used[i] = false;
	}
}
```
